\begin{frame}{Theoretical Issues }
  \begin{enumerate}
  \item Exposure Bias
    \begin{itemize}
    \item Training by conditioning on true $y_{1:t-1}$.
    \end{itemize}
    \air
  \item Metric Bias 
    \begin{itemize}
    \item Training with local NLL, evaluate with hamming-style losses.
    \end{itemize}

    \air
  \item Label Bias  %\cite{Lafferty2001}
    \begin{itemize}
    \item Locally normalized models have known pathological issues.
    \end{itemize}

  \end{enumerate}
\end{frame}

\begin{frame}{ Change 3: Replace train to enforce beam-search margin}

  New Global Objective: 
  \begin{itemize}

  \item Margin between gold seq $y^{(g)}$ and last seq on beam $y^{(K)}$  
  \end{itemize}
\begin{align*}
 \mathcal{L}&(\theta) = \sum_{t} \Delta(y^{(g)}_{1:t}, y_{1:t}^{K}) \left[1 - f(y^{(g)}_t, y^{(g)}_{1:t-1}, \cvec) +  f(y_t^{(K)}, y_{1:t-1}^{(K)}, \cvec) \right] 
\end{align*}

\begin{itemize}
  \item Slack-rescaled, margin-based sequence criterion, at each time step.  
    \air 

\item When violation occurs, target replaces current beam (learning as search optimization \cite{daume05learning})
\end{itemize}

\end{frame}

\begin{frame}{Sequence-Level Knowledge Distillation}
\air 
\air

Proposal: Replace multi-class with sequence distribution. Instead of word NLL, 

\begin{align*}
-\sum_t \sum_v &q(y_t=v \given y^{(g)}_{1: t-1}, \cvec ; \theta_T)\times  \log p(y_t =v \given y^{(g)}_{1: t-1}, \cvec ; \theta)
\end{align*}
 
% $$-\sum_t \log p(y_t=y_t \given y_{1:t-1}, \cvec ; \theta)$$

Minimize cross-entropy between $q$ and $p$ implied sequence-distribution 
\[
 -\sum_{v_1} \ldots \sum_{v_T} q(y_{1:T}= v_{1:T}  | \cvec; \theta_T) \times \log p(y_{1:T}=v_{1:T} | \cvec ; \theta)
\]
\air

However, as before this term is intractable.
\air
\air

\end{frame}

\begin{frame}{A Simple Approximation}
\air 
\air

%\centerline{Sequence-Level Knowledge Distillation}
\begin{columns}
\begin{column}{6cm}
Approximate $q(y_{1:T} \given \cvec )$ with mode
$$q(y_{1:T} \given \cvec ) \approx \mathbf{1}\{\argmax_{y} q(y_{1:T} \given \cvec )\}$$
\air
Roughly obtained with  beam search 
$$ y^*_{1:T} \approx  \argmax_{y_{1:T}} q(y_{1:T} \given \cvec ) $$
\\
Empirically, point estimate captures 
significant mass

\end{column}

\end{column}
\end{columns}
\end{frame}

\begin{frame}{Sequence-Level Knowledge Distillation}

\begin{columns}
\begin{column}{5cm}
\begin{align*}
\mathcal{L}_\text{SEQ-KD}(\theta) &= -\log p(y^*_{1:T} \given \cvec ; \theta)  \\
& \hspace*{-1cm} \displaystyle \approx   -\sum_{v_{1:T}} q(y_{1:T}=v_{1:t}| \cvec; \theta_T) \log p(y_{1:T} | \cvec ; \theta)
\end{align*}
Simplest model: train the student model on $y^*$ with NLL
